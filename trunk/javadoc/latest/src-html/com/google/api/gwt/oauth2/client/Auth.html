026     * Provides methods to manage authentication flow.

028     * @author jasonhall@google.com (Jason Hall)

032      /** Instance of the {@link Auth} to use in a GWT application. */
033      public static final Auth get() {

037      final TokenStore tokenStore;
038      private final Clock clock;
039      private final UrlCodex urlCodex;
040      final Scheduler scheduler;
041      String oauthWindowUrl;
042    
043      Auth(TokenStore tokenStore, Clock clock, UrlCodex urlCodex, Scheduler scheduler,
044          String oauthWindowUrl) {
045        this.tokenStore = tokenStore;
046        this.clock = clock;
047        this.urlCodex = urlCodex;
048        this.scheduler = scheduler;
049        this.oauthWindowUrl = oauthWindowUrl;
050      }
051    
052      private AuthRequest lastRequest;
053      private Callback<String, Throwable> lastCallback;
054    
055      private static final double TEN_MINUTES = 10 * 60 * 1000;
056    
057      /**
058       * Request an access token from an OAuth 2.0 provider.
059       *
060       * <p>
061       * If it can be determined that the user has already granted access, and the
062       * token has not yet expired, and that the token will not expire soon, the
063       * existing token will be passed to the callback.
064       * </p>
065       *
066       * <p>
067       * Otherwise, a popup window will be displayed which may prompt the user to
068       * grant access. If the user has already granted access the popup will
069       * immediately close and the token will be passed to the callback. If access
070       * hasn't been granted, the user will be prompted, and when they grant, the
071       * token will be passed to the callback.
072       * </p>
073       *
074       * @param req Request for authentication.
075       * @param callback Callback to pass the token to when access has been granted.
076       */
077      public void login(AuthRequest req, final Callback<String, Throwable> callback) {
078        lastRequest = req;
079        lastCallback = callback;
080    
081        String authUrl = req.toUrl(urlCodex) + "&redirect_uri=" + urlCodex.encode(oauthWindowUrl);
082    
083        // Try to look up the token we have stored.
084        final TokenInfo info = getToken(req);
085        if (info == null || info.expires == null || expiringSoon(info)) {
086          // Token wasn't found, or doesn't have an expiration, or is expired or
087          // expiring soon. Requesting access will refresh the token.
088          doLogin(authUrl, callback);
089        } else {
090          // Token was found and is good, immediately execute the callback with the
091          // access token.
092    
093          scheduler.scheduleDeferred(new ScheduledCommand() {
094            @Override
095            public void execute() {
096              callback.onSuccess(info.accessToken);
097            }
098          });
099        }
100      }
101    
102      /**
103       * Returns whether or not the token will be expiring within the next ten
104       * minutes.
105       */
106      boolean expiringSoon(TokenInfo info) {
107        // TODO(jasonhall): Consider varying the definition of "soon" based on the
108        // original expires_in value (e.g., "soon" = 1/10th of the total time before
109        // it's expired).
110        return Double.valueOf(info.expires) < (clock.now() + TEN_MINUTES);
111      }
112    
113      /**
114       * Get the OAuth 2.0 token for which this application may not have already
115       * been granted access, by displaying a popup to the user.
116       */
117      abstract void doLogin(String authUrl, Callback<String, Throwable> callback);
118    
119      /**
120       * Set the oauth window URL to use to authenticate. This will be provided by
121       * default, and is essentially only provided to enable authentication in the
122       * embedded Explorer.
123       */
124      public void setOAuthWindowUrl(String url) {
125        this.oauthWindowUrl = url;
126      }

128      /**
129       * Called by the {@code doLogin()} method which is registered as a global
130       * variable on the page.
131       */
132      // This method is called via a global method defined in AuthImpl.register()
133      @SuppressWarnings("unused")
134      void finish(String hash) {
135        TokenInfo info = new TokenInfo();
136        String error = null;
137        String errorDesc = "";
138        String errorUri = "";
139    
140        // Iterate over keys and values in the string hash value to find relevant
141        // information like the access token or an error message. The string will be
142        // in the form of: #key1=val1&key2=val2&key3=val3 (etc.)
143        int idx = 1;
144        while (idx < hash.length() - 1) {
145          // Grab the next key (between start and '=')
146          int nextEq = hash.indexOf('=', idx);
147          if (nextEq < 0) {
148            break;
149          }
150          String key = hash.substring(idx, nextEq);
151    
152          // Grab the next value (between '=' and '&')
153          int nextAmp = hash.indexOf('&', nextEq);
154          nextAmp = nextAmp < 0 ? hash.length() : nextAmp;
155          String val = hash.substring(nextEq + 1, nextAmp);
156    
157          // Start looking from here from now on.
158          idx = nextAmp + 1;
159    
160          // Store relevant values to be used later.
161          if (key.equals("access_token")) {
162            info.accessToken = val;
163          } else if (key.equals("expires_in")) {
164            // expires_in is seconds, convert to milliseconds and add to now
165            Double expiresIn = Double.valueOf(val) * 1000;
166            info.expires = String.valueOf(clock.now() + expiresIn);
167          } else if (key.equals("error")) {
168            error = val;
169          } else if (key.equals("error_description")) {
170            errorDesc = " (" + val + ")";
171          } else if (key.equals("error_uri")) {
172            errorUri = "; see: " + val;
173          }
174        }
175    
176        if (error != null) {
177          lastCallback.onFailure(
178              new RuntimeException("Error from provider: " + error + errorDesc + errorUri));
179        } else if (info.accessToken == null) {
180          lastCallback.onFailure(new RuntimeException("Could not find access_token in hash " + hash));
181        } else {
182          setToken(lastRequest, info);
183          lastCallback.onSuccess(info.accessToken);
184        }
185      }

187      /** Test-compatible abstraction for getting the current time. */
188      static interface Clock {
189        // Using double to avoid longs in GWT, which are slow.
190        double now();
191      }
192    
193      /** Test-compatible URL encoder/decoder. */
194      static interface UrlCodex {
195        /**
196         * URL-encode a string. This is abstract so that the Auth class can be
197         * tested.
198         */
199        String encode(String url);
200    
201        /**
202         * URL-decode a string. This is abstract so that the Auth class can be
203         * tested.
204         */
205        String decode(String url);
206      }
207    
208      TokenInfo getToken(AuthRequest req) {
209        String tokenStr = tokenStore.get(req.asString());
210        return tokenStr != null ? TokenInfo.fromString(tokenStr) : null;
211      }
212    
213      void setToken(AuthRequest req, TokenInfo info) {
214        tokenStore.set(req.asString(), info.asString());
215      }
216    
217      /**
218       * Clears all tokens stored by this class.
219       *
220       * <p>
221       * This will result in subsequent calls to
222       * {@link #login(AuthRequest, Callback)} displaying a popup to the user. If
223       * the user has already granted access, that popup will immediately close.
224       * </p>
225       */
226      public void clearAllTokens() {
227        tokenStore.clear();
228      }
229    
230      /** Encapsulates information an access token and when it will expire. */
231      static class TokenInfo {
232        String accessToken;
233        String expires;
234    
235        String asString() {
236          return accessToken + "-----" + (expires == null ? "" : expires);
237        }
238    
239        static TokenInfo fromString(String val) {
240          String[] parts = val.split("-----");
241          TokenInfo info = new TokenInfo();
242          info.accessToken = parts[0];
243          info.expires = parts.length > 1 ? parts[1] : null;
244          return info;
245        }
246      }
247    
248      /*
249       * @param req The authentication request of which to request the expiration
250       *        status.
251       * @return The number of milliseconds until the token expires, or negative
252       *         infinity if no token was found.
253       */
254      public double expiresIn(AuthRequest req) {
255        String val = tokenStore.get(req.asString());
256        return val == null ? Double.NEGATIVE_INFINITY :
257            Double.valueOf(TokenInfo.fromString(val).expires) - clock.now();
258      }

260      /**
261       * Exports a function to the page's global scope that can be called from regular JavaScript.
262       *
263       * <p>Usage (in JavaScript):</p>
264       * <code>
265       * oauth2.login({
266       *   "authUrl": "..." // the auth URL to use
267       *   "clientId": "..." // the client ID for this app
268       *   "scopes": ["...", "..."], // (optional) the scopes to request access to
269       *   "scopeDelimiter": "..." // (optional) the scope delimiter to use
270       * }, function(token) { // (optional) called on success, with the token
271       * }, function(error) { // (optional) called on error, with the error message
272       * });
273       * </code>
274       */
275      public static native void export() /*-{
276        if (!$wnd.oauth2) {
277          $wnd.oauth2 = {};
278        }
279        $wnd.oauth2.login = $entry(function(req, success, failure) {
280          @com.google.api.gwt.oauth2.client.Auth::nativeLogin(*)(req, success, failure);
281        });
282    
283        $wnd.oauth2.expiresIn = $entry(function(req) {
284          return @com.google.api.gwt.oauth2.client.Auth::nativeExpiresIn(*)(req);
285        });
286      }-*/;
287    
288      private static void nativeLogin(AuthRequestJso req, JsFunction success, JsFunction failure) {
289        AuthImpl.INSTANCE.login(fromJso(req), CallbackWrapper.create(success, failure));
290      }
291    
292      private static double nativeExpiresIn(AuthRequestJso req) {
293        return AuthImpl.INSTANCE.expiresIn(fromJso(req));
294      }
295    
296      private static AuthRequest fromJso(AuthRequestJso jso) {
297        return new AuthRequest(jso.getAuthUrl(), jso.getClientId())
298            .withScopes(jso.getScopes())
299            .withScopeDelimiter(jso.getScopeDelimiter());
300      }
301    
302      private static final class AuthRequestJso extends JavaScriptObject {
303        protected AuthRequestJso() {
304        }
305    
306        private final native String getAuthUrl() /*-{
307          return this.authUrl;
308        }-*/;
309    
310        private final native String getClientId() /*-{
311          return this.clientId;
312        }-*/;
313    
314        private final native JsArrayString getScopesNative() /*-{
315          return this.scopes || [];
316        }-*/;
317    
318        private final String[] getScopes() {
319          JsArrayString jsa = getScopesNative();
320          String[] arr = new String[jsa.length()];
321          for (int i = 0; i < jsa.length(); i++) {
322            arr[i] = jsa.get(i);
323          }
324          return arr;
325        }
326    
327        private final native String getScopeDelimiter() /*-{
328          return this.scopeDelimiter || " ";
329        }-*/;
330      }
331    
332      private static final class JsFunction extends JavaScriptObject {
333        protected JsFunction() {
334        }
335    
336        private final native void execute(String input) /*-{
337          this(input);
338        }-*/;
339      }
340    
341      private static final class CallbackWrapper implements Callback<String, Throwable> {
342        private final JsFunction success;
343        private final JsFunction failure;
344    
345        private CallbackWrapper(JsFunction success, JsFunction failure) {
346          this.success = success;
347          this.failure = failure;
348        }
349    
350        private static CallbackWrapper create(JsFunction success, JsFunction failure) {
351          return new CallbackWrapper(success, failure);
352        }
353    
354        @Override
355        public void onSuccess(String result) {
356          if (success != null) {
357            success.execute(result);
358          }
359        }
360    
361        @Override
362        public void onFailure(Throwable reason) {
363          if (failure != null) {
364            failure.execute(reason.getMessage());
365          }
366        }
367      }
